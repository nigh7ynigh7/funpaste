var timedKill = function(callback){
    var now = new Date();
    Paste.find()
        .where('removalDate')
            .lt(now.getTime())
        .exec(function(err,docs){
            if (err) {
                callback(true, err);
            }
            else {
                var remove = function(err, removed){
                    if (err){
                        return;
                    }
                };
                for(var i=0;i < docs.length;i++){
                    Paste.remove({_id: docs[i]._id}, remove);
                }
                callback(false, 'Old entries removed');
            }
        }
    );
};
